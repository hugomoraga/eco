class SimulationObserver(ABC):
    """Observer interface for simulation events. Read-only, no side effects."""

    def on_turn_start(self, turn: int, world: World) -> None: pass
    def on_turn_end(self, turn: int, world: World, action_taken: str | None) -> None: pass
    def on_action_selected(self, turn: int, action_name: str | None) -> None: pass
    def on_action_result(self, turn: int, action_name: str, result: ActionResult) -> None: pass
    def on_metric_changed(self, turn: int, metric: str, old_val: float, new_val: float) -> None: pass
    def on_event(self, turn: int, event_type: str, title: str, summary: str) -> None: pass
    def on_circle_founded(self, turn: int, circle_name: str, members: int) -> None: pass
    def on_npc_created(self, turn: int, npc_name: str, npc_role: str) -> None: pass
    def on_echo_spawned(self, turn: int, parent_name: str, daughter_name: str) -> None: pass
    def on_crisis(self, turn: int, metric: str, value: float) -> None: pass

### 1. console.py — Rich Console singleton

```python
from rich.console import Console as RichConsole
from rich.panel import Panel
from rich.table import Table
from rich.text import Text
from rich.live import Live

class Console:
    _instance = None

    def __init__(self):
        self._console = RichConsole()
        self._live: Live | None = None

    @classmethod
    def get(cls) -> "Console":
        if cls._instance is None:
            cls._instance = cls()
        return cls._instance

    def print(self, *args, **kwargs):
        self._console.print(*args, **kwargs)

    def print_panel(self, content: str | Text, title: str = "", border_style: str = "green"):
        self._console.print(Panel(content, title=title, border_style=border_style))

    def print_table(self, title: str, rows: list[list[str]], columns: list[str]):
        table = Table(title=title)
        for col in columns:
            table.add_column(col)
        for row in rows:
            table.add_row(*row)
        self._console.print(table)

    def print_divider(self, char="─", color="cyan"):
        self._console.print(f"[{color}]{char * 60}[/{color}]")

    def clear(self):
        self._console.clear()

    # ─── Live display (para autoplay) ───────────────────────────────

    def start_live(self, renderable):
        """Inicia display continuamente actualizable. Usar en autoplay."""
        self._live = Live(renderable, console=self._console, refresh_per_second=4)
        self._live.start()

    def update_live(self, renderable):
        if self._live:
            self._live.update(renderable)

    def stop_live(self):
        if self._live:
            self._live.stop()
            self._live = None

    def is_live(self) -> bool:
        return self._live is not None
```

### 5. input.py — CommandInput y COMMANDS registry

```python
class CommandInput:
    _session: PromptSession | None = None

    @classmethod
    def get_session(cls) -> PromptSession:
        if cls._session is None:
            cls._session = PromptSession(
                history=InMemoryHistory(),
                vi_mode=False,
                enable_history_search=True,
                message="► ",
            )
        return cls._session

    @classmethod
    def get_command(cls, prompt: str = "► ") -> str:
        """Get single line command. Returns empty string on EOF/Interrupt."""
        session = cls.get_session()
        try:
            return session.prompt(prompt).strip()
        except (EOFError, KeyboardInterrupt):
            return ""

    @classmethod
    def get_number(cls, prompt: str = "► ") -> int | None:
        """Get numeric input. Returns None for non-numeric."""
        cmd = cls.get_command(prompt)
        try:
            return int(cmd)
        except ValueError:
            return None

    @classmethod
    def get_choice(cls, options: list[str], prompt: str = "► ") -> int | None:
        """Get choice by number (1-indexed). Returns None for empty or out of range."""
        cmd = cls.get_command(prompt)
        if not cmd:
            return None
        try:
            idx = int(cmd) - 1
            if 0 <= idx < len(options):
                return idx
        except ValueError:
            pass
        return None

# Decorator para registrar comandos
def command(name: str):
    def decorator(func):
        COMMANDS[name] = func
        return func
    return decorator
```

**Validación de input:**
- `get_command()`: Devuelve string vacío en EOF/Interrupt (no lanza)
- `get_choice()`: Devuelve None si input vacío, no numérico, o fuera de rango
- No imprime mensajes de error — el caller decide qué mostrar

**Comandos predefinidos:**
- `/help` — muestra comandos disponibles
- `/history` — muestra historial de acciones
- `/replay [n]` — replay de últimas n acciones
- `/save [nombre]` — guarda partida
- `/load [nombre]` — carga partida
- `/quit` — salir del juego

## Interface — Punto de entrada del jugador

`ui_core/interface.py` coordina UI y jugador:

```python
class Interface:
    def __init__(self, world: Any, config: dict)

    def main_menu(self) -> str:
        """Muestra menú principal. Returns: 'new'|'continue'|'load'|'quit'"""

    def turn_loop(self, turn: int) -> dict[str, Any]:
        """Loop principal de un turno:
        1. Muestra header + metrics
        2. Muestra acciones disponibles
        3. Captura input (número, /comando, o q)
        4. Ejecuta y muestra resultado
        Returns: {"action": str, "result": str, "success": bool}
        """

    def show_event(self, event_type: str, title: str, summary: str = "")
    def show_replay(self, n: int = 5)
    def show_metrics_delta(self, old_metrics, new_metrics)
    def print_action_result(self, turn, action_type, message, details=None)
    def print_status_line(self, turn, world)

    def prompt_invalid_input(self, message: str = "Entrada inválida"):
        """Muestra error y re-pide input. Called por turn_loop cuando get_choice returns None."""
        self.console.print(f"[red]✗ {message}[/red]")
```

**Flujo de turn_loop con validación:**
```
turn_loop(turn):
  1. Mostrar header + metrics
  2. Mostrar acciones (numeradas)
  3. Loop hasta recibir input válido o 'q':
     a. cmd = get_command()
     b. Si startswith('/') → handle_command
     c. Si isdigit() → validar rango
        - Si válido: do_action
        - Si inválido (fuera de rango): prompt_invalid_input() + continue
     d. Si 'q': running=False, return {"action": None}
  4. Mostrar action_result
  5. Return {"action": ..., "result": ..., "success": ...}
```

## Flujo de un Turno (actualizado)

```
run_turn(turn):
  1. simulation.tick()  # Actualiza estado del mundo
  
  2. observers.notify("on_turn_start", turn, world)
     → ConsoleDisplay.render_turn_start()
        - turn_header
        - world_metrics_table
        - status_line
  
  3. player_mode.get_action(turn, world)
     → PlayerMode: Interface.turn_loop() con input validation
     → AutoplayMode: game_core.ai decide
     → HybridMode: AI sugiere + humano confirma
  
  4. observers.notify("on_action_selected", turn, action_name)
  
  5. try:
       result = action.execute(world)
     except Exception as e:
       observers.notify("on_error", turn, action_name, e)
       result = ActionResult(success=False, error=str(e))
  
  6. observers.notify("on_action_result", turn, action_name, result)
     → ConsoleDisplay.render_action_result()
        - action_result con emoji
        - action_details
  
  7. simulation.tick_post_action()
  
  8. events.evaluate(world)  # Posibles eventos aleatorios
  
  9. observers.notify("on_event", turn, event.type, event.title, event.summary)
     → ConsoleDisplay.render_event()
  
  10. observers.notify("on_turn_end", turn, world, action_name)
      → ConsoleDisplay.render_turn_end()
         - metrics_delta_table (si hay cambios)
         - status_line actualizado
  
  11. Si autoplay y no es live: breve delay antes del próximo turno
```

## Error Handling

```python
class Interface:
    def handle_action_error(self, action_name: str, error: Exception) -> None:
        """Muestra error de acción sin Crashear."""
        self.console.print(f"[red]✗ Error en {action_name}: {error}[/red]")
        self.console.print("[dim]La simulación continúa...[/dim]")

    def handle_critical_error(self, error: Exception) -> bool:
        """Error fatal. Pregunta si continuar o salir.
        Returns: True = continuar, False = salir"""
        self.console.print(f"[red]█ ERROR CRÍTICO: {error}[/red]")
        choice = CommandInput.get_choice(["Continuar", "Salir"])
        return choice == 0  # "Continuar"
```

**Regla:** Ningún error de acción mata el juego. Siempre se pregunta si continuar.
